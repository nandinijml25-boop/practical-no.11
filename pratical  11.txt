<div class="card"> 
<h3>Save Location</h3> 
<input type="text" id="username" placeholder="Enter your name" required title="User 
name"> 
<br> 
<button onclick="saveData()" title="Save data">Save</button> 
<button onclick="loadData()" title="Load data">Load</button> 
<button onclick="clearData()" title="Clear storage">Clear</button> 
<p id="output"></p> 
</div>

<script> 
/* Drag */ 
const dragItem = document.getElementById("dragItem"); 
const dropZone = document.getElementById("dropZone"); 
const locationText = document.getElementById("location"); 
dragItem.addEventListener("dragstart", (e)=>{ 
e.dataTransfer.setData("text/plain", "dragItem"); 
}); 
/* Allow drop */ 
dropZone.addEventListener("dragover", (e)=>{ 
e.preventDefault(); 
}); 
/* Drop → Get location */ 
dropZone.addEventListener("drop", (e)=>{ 

 
 
    e.preventDefault(); 
 
    if(navigator.geolocation){ 
        navigator.geolocation.getCurrentPosition((pos)=>{ 
            const lat = pos.coords.latitude; 
            const lon = pos.coords.longitude; 
 
            locationText.innerHTML = 
                "📍 Lat: " + lat + "<br>📍 Lon: " + lon; 
 
            // Save temporary 
            localStorage.setItem("lat", lat); 
            localStorage.setItem("lon", lon); 
 
        }, ()=>{ 
            locationText.innerHTML = "Permission denied!"; 
        }); 
    } else { 
        locationText.innerHTML = "Not supported!"; 
    } 
}); 
 
/* Storage */ 
function saveData(){ 
    const name = document.getElementById("username").value; 
 
    if(name === ""){ 
        alert("Enter name!"); 
        return; 
    } 
 
    localStorage.setItem("name", name); 
    alert("Saved Successfully!"); 
} 
 
function loadData(){ 

const name = localStorage.getItem("name"); 
const lat = localStorage.getItem("lat"); 
const lon = localStorage.getItem("lon"); 
document.getElementById("output").innerHTML = 
"👤 Name: " + name + "<br>" + 
"📍 Lat: " + lat + "<br>" + 
"📍 Lon: " + lon; 
} 
function clearData(){ 
localStorage.clear(); 
document.getElementById("output").innerHTML = "Storage Cleared"; 
}
